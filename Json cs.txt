<script>
  const fmtBRL = new Intl.NumberFormat('pt-BR', { style:'currency', currency:'BRL' });

  let DB = null;
  let charts = {};

  function destroyChart(id){
    if(charts[id]){ charts[id].destroy(); charts[id] = null; }
  }

  function buildBarChart(canvasId, labels, values){
    destroyChart(canvasId);
    const ctx = document.getElementById(canvasId);
    charts[canvasId] = new Chart(ctx, {
      type: 'bar',
      data: { labels, datasets: [{ data: values }] },
      options: {
        responsive:true,
        plugins:{ legend:{display:false}, tooltip:{enabled:true} },
        scales:{ x:{ ticks:{ color:'#9aa4c0' }, grid:{ color:'rgba(36,48,85,.4)' } },
                 y:{ ticks:{ color:'#9aa4c0' }, grid:{ color:'rgba(36,48,85,.4)' } } }
      }
    });
  }

  function buildDoughnut(canvasId, labels, values){
    destroyChart(canvasId);
    const ctx = document.getElementById(canvasId);
    charts[canvasId] = new Chart(ctx, {
      type: 'doughnut',
      data: { labels, datasets: [{ data: values }] },
      options: { responsive:true, plugins:{ legend:{ labels:{ color:'#9aa4c0' } } } }
    });
  }

  function buildLine(canvasId, labels, values){
    destroyChart(canvasId);
    const ctx = document.getElementById(canvasId);
    charts[canvasId] = new Chart(ctx, {
      type: 'line',
      data: { labels, datasets: [{ data: values, tension:.25 }] },
      options: {
        responsive:true,
        plugins:{ legend:{display:false} },
        scales:{ x:{ ticks:{ color:'#9aa4c0' }, grid:{ color:'rgba(36,48,85,.4)' } },
                 y:{ ticks:{ color:'#9aa4c0' }, grid:{ color:'rgba(36,48,85,.4)' } } }
      }
    });
  }

  function setKPIs(m){
    const k = DB.monthly[m].kpis;
    document.getElementById('kpiQtd').textContent = k.qtd_devolucoes ?? '—';
    document.getElementById('kpiValor').textContent = (k.valor_devolvido!=null) ? fmtBRL.format(k.valor_devolvido) : '—';
    document.getElementById('kpiPend').textContent = k.pendentes ?? '—';
    document.getElementById('kpiFin').textContent = k.finalizados ?? '—';
    document.getElementById('kpiTopMot').textContent = k.top_motivo?.label ?? '—';
    document.getElementById('kpiTopMotPct').textContent = (k.top_motivo?.pct!=null) ? (k.top_motivo.pct + '% do mês') : '—';
    document.getElementById('kpiTopSub').textContent = k.top_submotivo?.label ?? '—';
    document.getElementById('kpiTopSubPct').textContent = (k.top_submotivo?.pct!=null) ? (k.top_submotivo.pct + '% do mês') : '—';
  }

  function setCharts(m){
    const c = DB.monthly[m].charts;

    buildBarChart('chMotivo',
      c.motivo_macro.map(x=>x.label),
      c.motivo_macro.map(x=>x.value)
    );

    buildBarChart('chSub',
      c.submotivo_top5.map(x=>x.label),
      c.submotivo_top5.map(x=>x.value)
    );

    buildDoughnut('chStatus',
      c.status_ticket.map(x=>x.label),
      c.status_ticket.map(x=>x.value)
    );

    buildBarChart('chVend',
      c.vendedor_top5_valor.map(x=>x.label),
      c.vendedor_top5_valor.map(x=>x.value)
    );

    buildLine('chEvo',
      DB.evolution.map(x=>x.month),
      DB.evolution.map(x=>x.qtd)
    );
  }

  function setTable(m){
    const rows = DB.monthly[m].top_tickets || [];
    const tbody = document.getElementById('topTable');
    tbody.innerHTML = rows.slice(0,10).map(r => `
      <tr>
        <td>${r.created_date ?? ''}</td>
        <td><a href="${r.zendesk_url ?? '#'}" target="_blank">${r.ticket_id ?? ''}</a></td>
        <td>${r.nf ?? ''}</td>
        <td>${(r.valor!=null)? fmtBRL.format(r.valor): ''}</td>
        <td>${r.motivo_macro ?? ''}</td>
        <td>${r.submotivo ?? ''}</td>
        <td>${r.status_ticket ?? ''}</td>
        <td>${r.transportador ?? ''}</td>
      </tr>
    `).join('');
  }

  function renderMonth(m){
    setKPIs(m);
    setCharts(m);
    setTable(m);
  }

  async function load(){
    const r = await fetch(`data.json?v=${Date.now()}`, { cache:'no-store' });
    DB = await r.json();

    document.getElementById('updatedAt').textContent = DB.updated_at || '—';

    const sel = document.getElementById('monthSelect');
    sel.innerHTML = (DB.months||[]).map(m => `<option value="${m}">${m}</option>`).join('');

    // default: último mês disponível
    const defaultMonth = (DB.months && DB.months.length) ? DB.months[DB.months.length-1] : null;
    if(defaultMonth) sel.value = defaultMonth;

    sel.addEventListener('change', e => renderMonth(e.target.value));

    if(defaultMonth) renderMonth(defaultMonth);
  }

  load();
</script>
